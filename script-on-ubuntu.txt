/etc/nginx/sites-available/quiz.signalschool.ac.th

server {
    listen 80;
    server_name quiz.signalschool.ac.th;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name quiz.signalschool.ac.th;

    # SSL configuration (already set by certbot)
    ssl_certificate /etc/letsencrypt/live/quiz.signalschool.ac.th/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/quiz.signalschool.ac.th/privkey.pem;

    # เพิ่มการกำหนด MIME types
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # ขยายขนาดไฟล์ที่รองรับ
    client_max_body_size 10M;

    # แก้ไขการตั้งค่า root location
    root /var/www/html; # Path ที่เซิร์ฟเวอร์จะมองหาไฟล์ static

    # สำหรับ React SPA - ส่ง index.html สำหรับทุกเส้นทางที่ไม่ใช่ไฟล์จริง
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # ส่วนสำคัญ: กำหนดให้ส่ง index.html เมื่อไม่พบไฟล์
        try_files $uri $uri/ /index.html;
    }

    # จัดการไฟล์ static ให้ถูกต้อง
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, max-age=31536000";
        try_files $uri =404;
    }

    # API endpoints
    location /api {
        proxy_pass http://localhost:3001/api;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # phpMyAdmin
    location /phpmyadmin {
        proxy_pass http://localhost:8080;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
    }
}
